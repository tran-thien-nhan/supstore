MAIL_HOST=smtp.gmail.com
MAIL_PORT=587
MAIL_USERNAME=pipclup28061997@gmail.com //hoặc mail cá nhân đã được xác minh 2 bước

MAIL_FROM_ADDRESS="hello@example.com"
MAIL_FROM_NAME="${APP_NAME}"

4.
composer install

5.
php artisan key:generate

6.
php artisan migrate

7.
php artisan serve
